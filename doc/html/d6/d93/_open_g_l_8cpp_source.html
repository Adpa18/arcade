226 bool OpenGL::loadTexture(const std::string &sprite2D)
227 {
228  GLenum format;
229 
230  if (sprite2D.empty()) {
231  return (false);
232  }
233  if (this->tex[sprite2D] == 0) {
234  this->tex[sprite2D] = IMG_Load(std::string("./assets/sprites2D/" + sprite2D).c_str());
235  }
236  if (this->tex[sprite2D]->format->BytesPerPixel == 4) {
237  format = (this->tex[sprite2D]->format->Rmask == 0x000000ff) ? GL_RGBA : GL_BGRA;
238  } else if (this->tex[sprite2D]->format->BytesPerPixel == 3) {
239  format = (this->tex[sprite2D]->format->Rmask == 0x000000ff) ? GL_RGB : GL_BGR;
240  } else {
241  return (false);
242  }
243  glGenTextures(1, NULL);
244  glBindTexture(GL_TEXTURE_2D, 0);
245  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
246  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
247  glTexImage2D(GL_TEXTURE_2D, 0, this->tex[sprite2D]->format->BytesPerPixel,this->tex[sprite2D]->w,
248  this->tex[sprite2D]->h, 0, format, GL_UNSIGNED_BYTE, this->tex[sprite2D]->pixels);
249  return (true);
250 }
